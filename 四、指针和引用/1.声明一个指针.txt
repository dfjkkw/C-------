//声明一个指针
//指针定义：可以保存某个内存地址的变量
//野指针定义:指针指向的位置是不可知的（随机的、不正确的、没有明确限制的）指针变量
#include<iostream>
using namespace std;
int main()
{
	int a = 1, b = 1;
	int* p1, * p2;//p1和p2为野指针   注：若访问了野指针会导致程序的崩溃
	p1 = &a;//让指针p1指向a的地址
	p2 = &b;
	cout << "a的值：" << a << " " << "b的值：" << b << endl;
	cout << "a的地址：" << p1 << " " << "b的地址：" << p2 << endl;
	cout << "a的地址：" << &a << " " << "b的地址：" << &b << endl;
	//指针前加*可以输出该地址保存的数值
	cout << "p1的地址所保存的数值：" << *p1 << " " << "p2的地址所保存的数值：" << *p2 << endl;
	(*p1) += 5;
	cout << "a现在的值是：" << a<<endl;
	return 0;
}
