//队列(先进先出)
/*
空
________________________________
|______________________________|
队头                            队尾
压入元素
________________________________
|元素一_________________________|
队头                            队尾
压入元素
________________________________
|元素一，元素二__________________|
队头                            队尾
取出元素
________________________________
|元素二_________________________|
队头                            队尾
压入元素
________________________________
|元素二，元素三__________________|
队头                            队尾
取出元素
________________________________
|元素三_________________________|
队头                            队尾
取出元素
________________________________
|______________________________|
队头                            队尾
*/
#include<iostream>
#include<queue>
#include<time.h>
#include<string>
using namespace std;
int main()
{
	cout << "主妇又开始了新的一天，请你来为她安排工作任务吧！" << endl;
	string order;
	queue<string>task;
	srand(time(0));
	for (int i = 0;i < 8;i++)
	{
		if (i < 3||rand()&1)//判断是不是奇数
		{
			cout << "请为她安排一项新任务：";
			cin >> order;
			task.push(order);//为队列添加一个元素
		}
		else {
			if (task.empty())//判断队列是否为空
			{
				cout << "目前没有任务。" << endl;
			}
			else {
				cout << task.front()<<"已完成"<<endl;
				task.pop();//取走队头的元素
			}
		}
	}
	while (!task.empty())
	{
		cout << task.front() << "已完成" << endl;
		task.pop();//取走队头的元素
	}
	cout << "目前没有任务。" << endl;
	return 0;
}
