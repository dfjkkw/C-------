//建立学生成绩管理系统的编程思路
// 建立两个类，一个学生类，一个数据集类
// 学生类：
// 完成个人信息的输出和返回
class Student {
	SetMath();//输入数学成绩
	SetEnglish();//输入英语成绩
	SetComputer();//输入计算机成绩
	SetName();//设置姓名
	GetSum();//将学生总分返回至外界
	Match();//输入字符串信息与名字或学号是否匹配
	Print();//显示这个学生的信息
};
//数据集类：
// 将学生类的数组作为一个成员，封装对数组添加元素，修改元素，删除和查找等
class DataBase {
	Push();//新添加学生
	Search();//查找学生信息
	Delete();//删除一个学生
	Alter();//修改学生信息
	Display();//列表显示所有学生信息
	ShowMenu();//显示功能菜单
};
