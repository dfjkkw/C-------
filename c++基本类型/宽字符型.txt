//宽字符型
#include<iostream>
#include<string>
using namespace std;
enum level{POOR,MEDIUM,GOOD,EXCELLENCE};//枚举
int main()
{
	//枚举
	level my_score, my_score2, my_score3;
	my_score = MEDIUM;
	my_score2 = POOR;
	my_score3 = GOOD;
	cout << my_score << endl;
	cout << my_score2 << endl;
	cout << my_score3 << endl;
	//字符串
	string str1 = "我要", str2 = "自学网",str3 = str1 + str2;
	cout << str1 << endl;
	cout << str2 << endl;
	cout << str3 << endl;
	cout << str3[0] << str3[1] << endl;//接连两个字符输出一个汉字
	string str4 = "51zxw";
	cout << str4[0] << " ";
	cout << str4[1] << " ";
	cout << str4[2] << " ";
	cout << str4[3] << " ";
	cout << str4[4] << endl;
	//宽字符型wchar_t型变量用来保存汉字
	setlocale(LC_ALL, "chs");//输出汉字要先执行的函数
	wchar_t hz1, hz2;
	hz1 = L'中';
	hz2 = L'文';
	cout << hz1 << hz2 << endl;//cout不能输出宽字符型
	printf("如何输出中文呢？\n");
	printf("%wc%wc\n", hz1, hz2);//%wc输出宽字符型
	setlocale(LC_ALL,"C");//恢复原来的设置
	return 0;
}
